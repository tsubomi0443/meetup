+func GetMaxByColumn[T any](ctx context.Context, db *gorm.DB, columnName string) int64 {
+	var max sql.NullInt64
+	err := db.Model(new(T)).
+		Select(fmt.Sprintf("MAX(%s)", columnName)).
+		Take(&max).Error
+	if err != nil && err != gorm.ErrRecordNotFound {
+		return -1
+	}
+
+	if !max.Valid {
+		return -1
+	}
+	return max.Int64
